CREATE TABLE BOARD(
	BOARD_NUMBER INT AUTO_INCREMENT,
	CODE VARCHAR(10),
	TITLE VARCHAR(75),
	CONTENT VARCHAR(5000),
	WRITER VARCHAR(30),
	WRITE_DATE DATETIME,
	SEARCH_KEY VARCHAR(50),
	BOARD_USE VARCHAR(1),
	VIEW_CNT INT DEFAULT 0,
	PRIMARY KEY (BOARD_NUMBER)
);


CREATE TABLE USER(
	ID VARCHAR(20),
	PASSWORD VARCHAR(20),
	GENDER VARCHAR(1),
	NAME VARCHAR(20),
	AGE INT,
	EMAIL VARCHAR(30),
	PHONE_NUM VARCHAR(15),
	BIRTH DATE,
	AD_CHK VARCHAR(1),
	USER_USE VARCHAR(1),
	LEVEL INT,
	PRIMARY KEY(ID)
);

CREATE TABLE FILE(
	FILE_NUMBER INT AUTO_INCREMENT,
	BOARD_NUMBER INT,
	FILE_SIZE INT,
	FINE_NAME VARCHAR(100),
	FILE_USE VARCHAR(1),
	PRIMARY KEY(FILE_NUMBER),
	FOREIGN KEY(BOARD_NUMBER) REFERENCES BOARD(BOARD_NUMBER) ON UPDATE CASCADE ON DELETE CASCADE
);

CREATE TABLE COMMENT(
	COMMENT_NUMBER INT AUTO_INCREMENT,
	BOARD_NUMBER INT,
	COMMENT_WRITER VARCHAR(20),
	COMMENT_DATE DATETIME,
	COMMENT_CONTENT VARCHAR(2000),
	PRIMARY KEY(COMMENT_NUMBER),
	FOREIGN KEY(BOARD_NUMBER) REFERENCES BOARD(BOARD_NUMBER) ON UPDATE CASCADE ON DELETE CASCADE
);
